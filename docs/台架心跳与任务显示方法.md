## 概述

本文档描述测试台架看板系统中，台架在线状态（心跳）和正在测试项目的实现方法。

## 一、台架在线状态（心跳）

### 1.1 心跳机制说明

台架通过定期发送心跳请求来表明自己在线。系统根据心跳时间判断台架是否在线：

| 状态 | 判断条件 |
|-----|---------|
| **在线** | 最后心跳时间距今 ≤ 30秒 |
| **离线** | 最后心跳时间距今 > 30秒 或 从未发送心跳 |

### 1.2 心跳 API

**接口地址**: `POST /api/v1/benches/{bench_id}/heartbeat`

**请求体**:
```json
{
  "status": "running",        // 可选：台架状态 (running/offline/maintenance/alarm/idle)
  "current_task": "测试任务A", // 可选：当前执行的任务名称
  "metrics": {                // 可选：实时指标数据
    "temperature": 45.2,
    "rpm": 3000,
    "voltage": 12.5
  }
}
```

**响应**:
```json
{
  "message": "心跳成功",
  "timestamp": "2026-03-25T10:30:00.000Z"
}
```

### 1.3 台架端心跳发送示例

#### Python 示例

```python
import requests
import time
import json

class BenchClient:
    def __init__(self, bench_id: str, server_url: str = "http://localhost:8000"):
        self.bench_id = bench_id
        self.server_url = server_url
        self.heartbeat_url = f"{server_url}/api/v1/benches/{bench_id}/heartbeat"
    
    def send_heartbeat(self, status: str = "running", current_task: str = None, metrics: dict = None):
        """发送心跳"""
        payload = {
            "status": status,
            "current_task": current_task,
            "metrics": metrics
        }
        
        try:
            response = requests.post(self.heartbeat_url, json=payload, timeout=5)
            if response.status_code == 200:
                print(f"[心跳成功] {response.json()}")
                return True
            else:
                print(f"[心跳失败] {response.status_code}: {response.text}")
                return False
        except Exception as e:
            print(f"[心跳异常] {e}")
            return False
    
    def run_with_heartbeat(self, task_name: str, interval: int = 10):
        """执行任务并发送心跳"""
        print(f"开始执行任务: {task_name}")
        
        while True:
            # 发送心跳，状态为运行中
            self.send_heartbeat(
                status="running",
                current_task=task_name,
                metrics={
                    "progress": 50,  # 任务进度
                    "temperature": 45.0,
                    "runtime": 3600  # 运行时长(秒)
                }
            )
            time.sleep(interval)


# 使用示例
if __name__ == "__main__":
    # 台架ID（从看板获取或创建时保存）
    bench_id = "your-bench-id-here"
    
    client = BenchClient(bench_id)
    
    # 发送单次心跳
    client.send_heartbeat(
        status="running",
        current_task="HIL测试-电机控制器",
        metrics={"temperature": 42.5, "rpm": 2500}
    )
    
    # 或持续发送心跳
    # client.run_with_heartbeat("HIL测试-电机控制器", interval=10)
```

#### C/C++ 示例 (嵌入式设备)

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <curl/curl.h>
#include <json-c/json.h>

#define SERVER_URL "http://localhost:8000/api/v1/benches"
#define BENCH_ID "your-bench-id"
#define HEARTBEAT_INTERVAL 10  // 秒

// 发送心跳
int send_heartbeat(const char *status, const char *task, float temp, int rpm) {
    CURL *curl;
    CURLcode res;
    char url[256];
    char json_data[512];
    
    snprintf(url, sizeof(url), "%s/%s/heartbeat", SERVER_URL, BENCH_ID);
    
    // 构建JSON
    json_object *root = json_object_new_object();
    json_object_object_add(root, "status", json_object_new_string(status));
    if (task) {
        json_object_object_add(root, "current_task", json_object_new_string(task));
    }
    
    json_object *metrics = json_object_new_object();
    json_object_object_add(metrics, "temperature", json_object_new_double(temp));
    json_object_object_add(metrics, "rpm", json_object_new_int(rpm));
    json_object_object_add(root, "metrics", metrics);
    
    strcpy(json_data, json_object_to_json_string(root));
    json_object_put(root);
    
    // 发送HTTP POST
    curl = curl_easy_init();
    if (curl) {
        struct curl_slist *headers = NULL;
        headers = curl_slist_append(headers, "Content-Type: application/json");
        
        curl_easy_setopt(curl, CURLOPT_URL, url);
        curl_easy_setopt(curl, CURLOPT_POSTFIELDS, json_data);
        curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
        
        res = curl_easy_perform(curl);
        
        curl_slist_free_all(headers);
        curl_easy_cleanup(curl);
        
        return (res == CURLE_OK) ? 0 : -1;
    }
    return -1;
}

// 主循环
int main() {
    curl_global_init(CURL_GLOBAL_DEFAULT);
    
    while (1) {
        // 获取传感器数据
        float temperature = read_temperature();
        int rpm = read_rpm();
        
        // 发送心跳
        if (send_heartbeat("running", "电机耐久测试", temperature, rpm) == 0) {
            printf("[心跳成功]\n");
        } else {
            printf("[心跳失败]\n");
        }
        
        sleep(HEARTBEAT_INTERVAL);
    }
    
    curl_global_cleanup();
    return 0;
}
```

### 1.4 状态枚举值

| 状态值 | 说明 | 颜色 |
|-------|------|------|
| `running` | 运行中 | 🟢 绿色 |
| `idle` | 空闲 | 🔵 蓝色 |
| `offline` | 离线 | ⚪ 灰色 |
| `maintenance` | 维护中 | 🟡 黄色 |
| `alarm` | 告警 | 🔴 红色 |

### 1.5 后端心跳处理逻辑

后端收到心跳后执行以下操作：

1. 更新 `last_heartbeat` 时间戳
2. 如果提供了 `status`，更新台架状态
3. 如果提供了 `current_task`，更新当前任务
4. 如果提供了 `metrics`，更新指标数据
5. 通过 WebSocket 广播状态更新

```python
# backend/app/api/benches.py

@router.post("/{bench_id}/heartbeat")
async def heartbeat(bench_id: str, heartbeat_data: HeartbeatData, db: Session = Depends(get_db)):
    bench = db.query(TestBench).filter(TestBench.id == bench_id).first()
    if not bench:
        raise HTTPException(status_code=404, detail="台架不存在")
    
    # 维护中忽略心跳
    if bench.is_under_maintenance:
        return {"message": "台架在维护中，心跳被忽略"}
    
    # 更新心跳时间
    bench.last_heartbeat = datetime.utcnow()
    
    # 更新状态
    if heartbeat_data.status:
        bench.status = BenchStatus(heartbeat_data.status.value)
    elif bench.status == BenchStatus.OFFLINE:
        bench.status = BenchStatus.IDLE  # 从离线变为空闲
    
    # 更新任务
    if heartbeat_data.current_task:
        if bench.current_task != heartbeat_data.current_task:
            bench.current_task = heartbeat_data.current_task
            bench.task_start_time = datetime.utcnow()
    else:
        bench.current_task = None
        bench.task_start_time = None
    
    # 更新指标
    if heartbeat_data.metrics:
        bench.metrics = heartbeat_data.metrics
    
    db.commit()
    
    # WebSocket 广播
    await ws_manager.broadcast({
        "event": "bench_heartbeat",
        "data": {
            "id": bench_id,
            "status": bench.to_dict()["status"],
            "currentTask": bench.current_task,
            "lastHeartbeat": bench.last_heartbeat.isoformat()
        }
    })
    
    return {"message": "心跳成功", "timestamp": bench.last_heartbeat.isoformat()}
```

### 1.6 离线检测

后端设备监控服务定期检查心跳超时：

```python
# backend/app/core/device_monitor.py

class DeviceMonitor:
    def __init__(self):
        self.heartbeat_timeout = 30  # 心跳超时阈值（秒）
    
    def check_heartbeat_timeout(self, bench: TestBench) -> bool:
        """检查心跳是否超时"""
        if bench.last_heartbeat is None:
            return True  # 从未发送心跳，视为离线
        
        elapsed = (datetime.utcnow() - bench.last_heartbeat).total_seconds()
        return elapsed > self.heartbeat_timeout
```

## 三、完整示例

### 3.1 台架端完整实现 (Python)

```python
#!/usr/bin/env python3
"""
台架客户端 - 完整示例
实现心跳发送和任务状态管理
"""

import requests
import time
import threading
from datetime import datetime


class TestBenchClient:
    """台架客户端"""
    
    def __init__(self, bench_id: str, server_url: str = "http://localhost:8000"):
        self.bench_id = bench_id
        self.server_url = server_url
        self.heartbeat_url = f"{server_url}/api/v1/benches/{bench_id}/heartbeat"
        
        self._running = False
        self._current_task = None
        self._heartbeat_thread = None
        self._heartbeat_interval = 10  # 秒
    
    def start_heartbeat(self):
        """启动心跳线程"""
        if self._running:
            return
        
        self._running = True
        self._heartbeat_thread = threading.Thread(target=self._heartbeat_loop, daemon=True)
        self._heartbeat_thread.start()
        print("[客户端] 心跳线程已启动")
    
    def stop_heartbeat(self):
        """停止心跳"""
        self._running = False
        if self._heartbeat_thread:
            self._heartbeat_thread.join(timeout=5)
        print("[客户端] 心跳线程已停止")
    
    def _heartbeat_loop(self):
        """心跳循环"""
        while self._running:
            try:
                self._send_heartbeat()
            except Exception as e:
                print(f"[心跳异常] {e}")
            time.sleep(self._heartbeat_interval)
    
    def _send_heartbeat(self):
        """发送心跳请求"""
        payload = {
            "status": "running" if self._current_task else "idle",
            "current_task": self._current_task,
            "metrics": self._collect_metrics()
        }
        
        response = requests.post(self.heartbeat_url, json=payload, timeout=5)
        if response.status_code == 200:
            print(f"[心跳成功] {datetime.now().strftime('%H:%M:%S')} - 任务: {self._current_task or '空闲'}")
        else:
            print(f"[心跳失败] {response.status_code}")
    
    def _collect_metrics(self) -> dict:
        """收集实时指标（示例）"""
        return {
            "temperature": 42.5,
            "runtime": time.time(),
            "memory_usage": 65.2
        }
    
    def start_task(self, task_name: str):
        """开始任务"""
        self._current_task = task_name
        print(f"[任务开始] {task_name}")
        
        # 立即发送一次心跳
        self._send_heartbeat()
    
    def finish_task(self, result: str = "完成"):
        """结束任务"""
        print(f"[任务结束] {self._current_task} - {result}")
        self._current_task = None
        
        # 发送空闲状态心跳
        self._send_heartbeat()
    
    def run_test(self, task_name: str, duration: int = 60):
        """执行测试任务（示例）"""
        self.start_task(task_name)
        
        try:
            # 模拟测试执行
            for i in range(duration):
                if not self._running:
                    break
                print(f"  测试进度: {i+1}/{duration}秒")
                time.sleep(1)
            
            self.finish_task("成功")
        except KeyboardInterrupt:
            self.finish_task("中断")


# 使用示例
if __name__ == "__main__":
    # 配置
    BENCH_ID = "your-bench-id"  # 替换为实际台架ID
    SERVER_URL = "http://localhost:8000"
    
    # 创建客户端
    client = TestBenchClient(BENCH_ID, SERVER_URL)
    
    # 启动心跳
    client.start_heartbeat()
    
    try:
        # 执行测试任务
        client.run_test("HIL测试-电机控制器V2.0", duration=30)
        
        # 空闲一段时间
        print("等待新任务...")
        time.sleep(20)
        
        # 执行另一个任务
        client.run_task("耐久性测试-1000小时", duration=30)
        
    except KeyboardInterrupt:
        print("\n用户中断")
    finally:
        client.stop_heartbeat()
```

## 五、最佳实践

### 5.1 心跳间隔建议

| 场景 | 建议间隔 |
|-----|---------|
| 高频监控 | 5-10秒 |
| 常规监控 | 10-30秒 |
| 低功耗设备 | 30-60秒 |

### 5.2 任务命名规范

建议使用格式：`{测试类型}-{项目/产品名称}`

示例：
- `HIL测试-电机控制器V2.0`
- `耐久性测试-电池包`
- `功能测试-ADAS控制器`

### 5.3 指标数据建议

常用指标：
- `progress`: 任务进度 (0-100)
- `temperature`: 温度
- `runtime_seconds`: 运行时长
- `passed_cases`: 通过用例数
- `failed_cases`: 失败用例数
- `error_count`: 错误计数

## 六、故障排查

### 6.1 台架显示离线

检查项：
1. 心跳请求是否正常发送
2. 网络连接是否正常
3. 后端服务是否运行
4. 台架ID是否正确

### 6.2 任务不显示

检查项：
1. 心跳请求中是否包含 `current_task` 字段
2. WebSocket 连接是否正常
3. 前端是否正确处理 `bench_heartbeat` 事件
